##### Solution
```
def prime_numbers(n):
    primes = [True for i in range(n + 1)]
    l = 2
    while(l * l <= n):
        if(primes[l] == True):
            for i in range(l * l, n+1, l):
                primes[i] = False
        l += 1
    for i in range(2,n+1):
        if primes[i]:
            print(i)

n = int(input("Enter the value: "))
print("Following are the prime numbers smaller\nthan or equal to", n)
prime_numbers(n)
```
